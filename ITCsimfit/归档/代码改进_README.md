```r
# 安全执行

```

## 🔍 改进前后对比

### 对比1：参数边界定义

```r
# ❌ 改进前（重复、分散、硬编码）
# ui.R
sliderInput("logK1", "logK1", 1, 9, 7.0, ...)

# server.R
for(nm in names(par_vec)) {
  if(grepl("logK", nm)) { lower_b[nm] <- 1; upper_b[nm] <- 9 }
  if(grepl("H", nm))    { lower_b[nm] <- -15000; upper_b[nm] <- 5000 }
  # ... 更多重复
}

# ✅ 改进后（统一、集中、常量）
# constants.R（集中定义）
PARAM_BOUNDS <- list(
  logK = c(lower = 1, upper = 9),
  H = c(lower = -15000, upper = 5000)
)

# ui.R
sliderInput("logK1", "logK1", 
           min = PARAM_BOUNDS$logK["lower"],
           max = PARAM_BOUNDS$logK["upper"], ...)

# server.R
bounds <- get_parameter_bounds(names(par_vec), v_inj)
```

### 对比2：错误处理

```r
# ❌ 改进前（不统一）
tryCatch({ ... }, error = function(e) NULL)
tryCatch({ ... }, error = function(e) print(e))
tryCatch({ ... }, error = function(e) showNotification(...))

# ✅ 改进后（统一）
safe_execute({ ... }, context = "...", lang_val = lang())
# 自动：错误处理 + 日志记录 + 用户通知
```

### 对比3：数值检查

```r
# ❌ 改进前（魔法数字）
if(G_tot <= 1e-20) ...
log(pmax(x, 1e-15))

# ✅ 改进后（语义化）
if(is_near_zero(G_tot, EPSILON)) ...
safe_log(x, EPSILON_LOG)
```
